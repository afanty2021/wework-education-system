```python
# backend/routes/course.py
from flask import Blueprint, request, jsonify
from models import Course, db

course_bp = Blueprint('course', __name__)

@course_bp.route('/api/courses', methods=['GET'])
def get_courses():
    courses = Course.query.all()
    return jsonify([{
        'id': c.id,
        'name': c.name,
        'duration': c.duration
    } for c in courses])

@course_bp.route('/api/courses', methods=['POST'])
def create_course():
    data = request.json
    course = Course(
        name=data['name'],
        duration=data.get('duration', 60)
    )
    db.session.add(course)
    db.session.commit()
    return jsonify({'id': course.id, 'message': '创建成功'})
```

```python
# utils/wework_message.py
import requests
import json

class WeworkMessage:
    def __init__(self, corp_id, corp_secret, agent_id):
        self.corp_id = corp_id
        self.corp_secret = corp_secret
        self.agent_id = agent_id
        self.access_token = None

    def get_access_token(self):
        url = f"https://qyapi.weixin.qq.com/cgi-bin/gettoken"
        params = {
            'corpid': self.corp_id,
            'corpsecret': self.corp_secret
        }
        res = requests.get(url, params=params)
        self.access_token = res.json()['access_token']
        return self.access_token

    def send_text(self, user_id, content):
        url = f"https://qyapi.weixin.qq.com/cgi-bin/message/send"
        params = {'access_token': self.access_token}
        data = {
            "touser": user_id,
            "msgtype": "text",
            "agentid": self.agent_id,
            "text": {
                "content": content
            }
        }
        return requests.post(url, params=params, json=data)

# 使用示例
wework = WeworkMessage('your_corp_id', 'your_secret', 'your_agent_id')
wework.get_access_token()
wework.send_text('user_id', '您有一节课即将开始')
```

2. **定时任务**
```python
# tasks/reminder.py
from apscheduler.schedulers.background import BackgroundScheduler
from utils.wework_message import WeworkMessage
from models import Schedule, User, db
import datetime

scheduler = BackgroundScheduler()

def send_class_reminder():
    """发送上课提醒"""
    now = datetime.datetime.now()
    # 查询30分钟后的课程
    future_time = now + datetime.timedelta(minutes=30)

    schedules = Schedule.query.filter(
        Schedule.start_time.between(future_time - datetime.timedelta(minutes=5),
                                   future_time + datetime.timedelta(minutes=5))
    ).all()

    wework = WeworkMessage()
    wework.get_access_token()

    for schedule in schedules:
        # 给教师发送提醒
        content = f"上课提醒：{schedule.course.name}\n时间：{schedule.start_time}\n教室：{schedule.classroom}"
        wework.send_text(schedule.teacher.wework_id, content)

# 添加定时任务
scheduler.add_job(send_class_reminder, 'interval', minutes=5)
scheduler.start()
```
