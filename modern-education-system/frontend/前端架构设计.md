## 3. 核心功能模块

### 3.1 认证授权模块

```typescript
// src/hooks/useAuth.ts
import { ref, computed } from 'vue'
import { useAuthStore } from '@/stores/modules/auth'

export function useAuth() {
  const authStore = useAuthStore()

  const isAuthenticated = computed(() => authStore.isAuthenticated)
  const user = computed(() => authStore.user)
  const permissions = computed(() => authStore.permissions)

  const login = async (credentials: LoginParams) => {
    try {
      await authStore.login(credentials)
      return { success: true }
    } catch (error) {
      return { success: false, error }
    }
  }

  const logout = () => {
    authStore.logout()
  }

  const hasPermission = (permission: string) => {
    return permissions.value.includes(permission)
  }

  return {
    isAuthenticated,
    user,
    login,
    logout,
    hasPermission
  }
}
```

### 3.2 表格数据管理

```typescript
// src/hooks/useTable.ts
import { ref, reactive, computed } from 'vue'
import type { TableProps } from 'element-plus'

export function useTable<T = any>(api: Function) {
  const loading = ref(false)
  const data = ref<T[]>([])
  const total = ref(0)

  const queryParams = reactive({
    page: 1,
    size: 20,
    keyword: ''
  })

  const pagination = computed(() => ({
    currentPage: queryParams.page,
    pageSize: queryParams.size,
    total: total.value,
    layout: 'total, sizes, prev, pager, next, jumper'
  }))

  const loadData = async () => {
    loading.value = true
    try {
      const { items, total: totalCount } = await api(queryParams)
      data.value = items
      total.value = totalCount
    } finally {
      loading.value = false
    }
  }

  const handlePageChange = (page: number) => {
    queryParams.page = page
    loadData()
  }

  const handleSizeChange = (size: number) => {
    queryParams.size = size
    queryParams.page = 1
    loadData()
  }

  const refresh = () => {
    queryParams.page = 1
    loadData()
  }

  return {
    loading,
    data,
    pagination,
    queryParams,
    loadData,
    handlePageChange,
    handleSizeChange,
    refresh
  }
}
```

    :action="uploadUrl"
    :headers="headers"

    :on-success="handleSuccess"
    :on-error="handleError"
    :show-file-list="false"

import { computed } from 'vue'
import { Plus } from '@element-plus/icons-vue'
import { ElMessage } from 'element-plus'
import { useAuthStore } from '@/stores/modules/auth'

interface Props {
  modelValue: string
}

interface Emits {
  (e: 'update:modelValue', value: string): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const authStore = useAuthStore()

const uploadUrl = import.meta.env.VITE_API_URL + '/v1/files/upload'
const headers = computed(() => ({
  Authorization: `Bearer ${authStore.token}`
}))

const beforeUpload = (file: File) => {
  const isImage = file.type.startsWith('image/')
  const isLt2M = file.size / 1024 / 1024 < 2

  if (!isImage) {
    ElMessage.error('只能上传图片文件')
    return false
  }

  if (!isLt2M) {
    ElMessage.error('图片大小不能超过2MB')
    return false
  }

  return true
}

const handleSuccess = (response: any) => {
  if (response.code === 200) {
    emit('update:modelValue', response.data.url)
    ElMessage.success('上传成功')
  } else {
    ElMessage.error(response.message)
  }
}

const handleError = () => {
  ElMessage.error('上传失败')
}
</script>

## 4. 状态管理

### 4.1 用户状态管理

```typescript
// src/stores/modules/auth.ts
import { defineStore } from 'pinia'
import { login, logout, getUserInfo } from '@/api/auth'
import { setToken, removeToken } from '@/utils/auth'

interface User {
  id: string
  username: string
  realName: string
  avatar?: string
  roles: string[]
  permissions: string[]
}

interface AuthState {
  token: string
  user: User | null
}

export const useAuthStore = defineStore('auth', {
  state: (): AuthState => ({
    token: '',
    user: null
  }),

  getters: {
    isAuthenticated: (state) => !!state.token,
    roles: (state) => state.user?.roles || [],
    permissions: (state) => state.user?.permissions || []
  },

  actions: {
    async login(loginForm: LoginForm) {
      try {
        const { token } = await login(loginForm)
        this.token = token
        setToken(token)
        await this.getUserInfo()
      } catch (error) {
        throw error
      }
    },

    async getUserInfo() {
      const user = await getUserInfo()
      this.user = user
      return user
    },

    logout() {
      this.token = ''
      this.user = null
      removeToken()
      logout()
    }
  }
})
```

### 5.2 路由守卫

```typescript
// src/router/guards.ts
import router from '@/router'
import { useAuthStore } from '@/stores/modules/auth'
import { ElMessage } from 'element-plus'

router.beforeEach(async (to, from, next) => {
  const authStore = useAuthStore()

  // 设置页面标题
  document.title = to.meta.title ? `${to.meta.title} - 教务系统` : '教务系统'

  // 检查是否需要认证
  if (to.meta.requiresAuth !== false && !authStore.isAuthenticated) {
    next('/login')
    return
  }

  // 已经登录不能访问登录页
  if (to.path === '/login' && authStore.isAuthenticated) {
    next('/')
    return
  }

  // 检查权限
  if (to.meta.permission && !authStore.permissions.includes(to.meta.permission)) {
    ElMessage.error('没有访问权限')
    next('/403')
    return
  }

  next()
})
```

    <!-- 搜索栏 -->
    <view class="search-bar">
      <u-search

      >

      :status="loadMoreStatus"

<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { getMemberList } from '@/api/member'

const keyword = ref('')
const members = ref([])
const page = ref(1)
const loadMoreStatus = ref('loadmore')

const loadData = async () => {
  const res = await getMemberList({
    page: page.value,
    keyword: keyword.value
  })

  if (page.value === 1) {
    members.value = res.items
  } else {
    members.value.push(...res.items)

const handleSearch = () => {
  page.value = 1
  loadData()
}

  page.value++
  loadData()
}

const goToDetail = (id: string) => {
  uni.navigateTo({
    url: `/pages/member/detail?id=${id}`
  })
}

onMounted(() => {
  loadData()
})
