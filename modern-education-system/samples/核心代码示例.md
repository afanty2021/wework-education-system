```java
    @ExceptionHandler(BusinessException.class)
    public Result<Void> handleBusinessException(BusinessException e) {
        log.error("业务异常: {}", e.getMessage());
        return Result.error(e.getCode(), e.getMessage());
    }

    @ExceptionHandler(ValidationException.class)
    public Result<Void> handleValidationException(ValidationException e) {
        log.error("参数验证异常: {}", e.getMessage());
        return Result.error(10001, e.getMessage());
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public Result<Void> handleValidationException(MethodArgumentNotValidException e) {
        String message = e.getBindingResult().getFieldErrors().stream()
                .map(FieldError::getDefaultMessage)
                .collect(Collectors.joining(", "));
        return Result.error(10001, message);
    }

    @ExceptionHandler(Exception.class)
    public Result<Void> handleException(Exception e) {
        log.error("系统异常", e);
        return Result.error(50000, "系统繁忙，请稍后重试");
    }
}
```

```java
@Component
public class IdGenerator {

    private final Snowflake snowflake;

    public IdGenerator() {
        // 使用机器ID和数据中心ID配置雪花算法
        long workerId = getWorkerId();
        long datacenterId = getDatacenterId();
        this.snowflake = new Snowflake(workerId, datacenterId);
    }

    public String nextId() {
        return snowflake.nextIdStr();
    }

    private long getWorkerId() {
        try {
            String ip = InetAddress.getLocalHost().getHostAddress();
            return Math.abs(ip.hashCode()) % 32;
        } catch (Exception e) {
            return 1L;
        }
    }

    private long getDatacenterId() {
        return 1L; // 默认数据中心ID
    }
}
```

```java
// member-service/src/main/java/com/edu/member/service/impl/MemberServiceImpl.java
@Service
@Transactional
@Slf4j
public class MemberServiceImpl implements MemberService {

    @Autowired
    private MemberMapper memberMapper;

    @Autowired
    private MemberCardMapper memberCardMapper;

    @Autowired
    private IdGenerator idGenerator;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Autowired
    private MessageProducer messageProducer;

    @Override
    public MemberDTO createMember(CreateMemberCommand command) {
        // 1. 验证手机号唯一性
        if (memberMapper.existsByMobile(command.getMobile())) {
            throw new BusinessException(30001, "手机号已存在");
        }

        // 2. 创建会员
        Member member = new Member();
        member.setMemberId(idGenerator.nextId());
        member.setMemberNo(generateMemberNo());
        member.setRealName(command.getRealName());
        member.setMobile(command.getMobile());
        member.setGender(command.getGender());
        member.setBirthday(command.getBirthday());
        member.setSourceType(command.getSourceType());
        member.setSourceDetail(command.getSourceDetail());
        member.setSalesUserId(command.getSalesUserId());
        member.setMemberLevel(1);
        member.setPoints(0);
        member.setBalance(BigDecimal.ZERO);
        member.setStatus(1);

        memberMapper.insert(member);

        // 3. 发送欢迎消息
        MemberWelcomeEvent event = new MemberWelcomeEvent();
        event.setMemberId(member.getMemberId());
        event.setMobile(member.getMobile());
        event.setRealName(member.getRealName());
        messageProducer.send(event);

        // 4. 清除缓存
        clearMemberCache(member.getMemberId());

        log.info("创建会员成功: memberId={}, mobile={}", member.getMemberId(), member.getMobile());

        return MemberConverter.toDTO(member);
    }

    @Override
    @Cacheable(value = "member", key = "#memberId")
    public MemberDTO getMemberById(String memberId) {
        Member member = memberMapper.selectById(memberId);
        if (member == null) {
            throw new BusinessException(10002, "会员不存在");
        }
        return MemberConverter.toDTO(member);
    }

    @Override
    public PageResult<MemberDTO> getMemberList(MemberQuery query) {
        Page<Member> page = new Page<>(query.getPage(), query.getSize());

        LambdaQueryWrapper<Member> wrapper = new LambdaQueryWrapper<>();
        wrapper.like(StringUtils.isNotBlank(query.getKeyword()),
                Member::getRealName, query.getKeyword())
               .or()
               .like(StringUtils.isNotBlank(query.getKeyword()),
                Member::getMobile, query.getKeyword())
               .eq(query.getStatus() != null, Member::getStatus, query.getStatus())
               .orderByDesc(Member::getCreatedAt);

        Page<Member> result = memberMapper.selectPage(page, wrapper);

        return PageResult.of(result, MemberConverter::toDTO);
    }

    @Override
    public MemberCardDTO issueCard(IssueCardCommand command) {
        // 1. 验证会员状态
        Member member = memberMapper.selectById(command.getMemberId());
        if (member == null || member.getStatus() != 1) {
            throw new BusinessException(30002, "会员状态异常");
        }

        // 2. 检查卡类型
        CardType cardType = cardTypeMapper.selectById(command.getCardTypeId());
        if (cardType == null) {
            throw new BusinessException(30003, "卡类型不存在");
        }

        // 3. 生成会员卡
        MemberCard card = new MemberCard();
        card.setCardId(idGenerator.nextId());
        card.setCardNo(generateCardNo());
        card.setMemberId(command.getMemberId());
        card.setCardTypeId(command.getCardTypeId());
        card.setTotalHours(cardType.getHours());
        card.setUsedHours(BigDecimal.ZERO);
        card.setGiftHours(cardType.getGiftHours());
        card.setExpiredDate(calculateExpiredDate(cardType.getValidityMonths()));
        card.setStatus(1);

        memberCardMapper.insert(card);

        // 4. 发送会员卡事件
        CardIssuedEvent event = new CardIssuedEvent();
        event.setCardId(card.getCardId());
        event.setMemberId(card.getMemberId());
        event.setCardNo(card.getCardNo());
        messageProducer.send(event);

        return MemberCardConverter.toDTO(card);
    }

    private String generateMemberNo() {
        String date = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd"));
        String key = "member:no:" + date;
        Long count = redisTemplate.opsForValue().increment(key);
        redisTemplate.expire(key, Duration.ofDays(1));
        return "M" + date + String.format("%04d", count);
    }

    private String generateCardNo() {
        String date = LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyMMddHHmmss"));
        String random = String.format("%04d", ThreadLocalRandom.current().nextInt(10000));
        return "C" + date + random;
    }

    private LocalDate calculateExpiredDate(Integer validityMonths) {
        return LocalDate.now().plusMonths(validityMonths);
    }

    @CacheEvict(value = "member", key = "#memberId")
    private void clearMemberCache(String memberId) {
        // 缓存清除由@CacheEvict注解处理
    }
}
```

```java
// order-service/src/main/java/com/edu/order/service/impl/OrderServiceImpl.java
@Service
@Transactional
public class OrderServiceImpl implements OrderService {

    @Autowired
    private OrderMapper orderMapper;

    @Autowired
    private OrderDetailMapper orderDetailMapper;

    @Autowired
    private MemberCardService memberCardService;

    @Autowired
    private ProductService productService;

    @GlobalTransactional
    @Override
    public OrderDTO createOrder(CreateOrderCommand command) {
        // 1. 创建订单
        Order order = new Order();
        order.setOrderId(idGenerator.nextId());
        order.setOrderNo(generateOrderNo());
        order.setMemberId(command.getMemberId());
        order.setOrderType(command.getOrderType());
        order.setStatus(1); // 待支付
        order.setTotalAmount(BigDecimal.ZERO);
        order.setPaidAmount(BigDecimal.ZERO);

        // 2. 计算订单金额并创建订单明细
        List<OrderDetail> details = new ArrayList<>();
        BigDecimal totalAmount = BigDecimal.ZERO;

        for (CreateOrderCommand.Item item : command.getItems()) {
            // 获取产品信息
            ProductDTO product = productService.getProduct(item.getProductId());

            OrderDetail detail = new OrderDetail();
            detail.setDetailId(idGenerator.nextId());
            detail.setOrderId(order.getOrderId());
            detail.setProductType(item.getProductType());
            detail.setProductId(item.getProductId());
            detail.setProductName(product.getName());
            detail.setUnitPrice(product.getPrice());
            detail.setQuantity(item.getQuantity());
            detail.setTotalPrice(product.getPrice().multiply(new BigDecimal(item.getQuantity())));

            details.add(detail);
            totalAmount = totalAmount.add(detail.getTotalPrice());

            // 3. 扣减库存
            productService.reduceStock(item.getProductId(), item.getQuantity());
        }

        order.setTotalAmount(totalAmount);
        orderMapper.insert(order);

        // 4. 批量插入订单明细
        orderDetailMapper.insertBatch(details);

        // 5. 如果是会员卡订单，预锁定卡
        if (command.getOrderType() == 2) { // 会员卡订单
            for (CreateOrderCommand.Item item : command.getItems()) {
                if ("card".equals(item.getProductType())) {
                    memberCardService.lockCard(item.getProductId());
                }
            }
        }

        // 6. 发送订单创建事件
        OrderCreatedEvent event = new OrderCreatedEvent();
        event.setOrderId(order.getOrderId());
        event.setOrderNo(order.getOrderNo());
        event.setMemberId(order.getMemberId());
        event.setTotalAmount(totalAmount);
        messageProducer.send(event);

        return OrderConverter.toDTO(order);
    }

    @GlobalTransactional
    @Override
    public void payOrder(PayOrderCommand command) {
        // 1. 获取订单
        Order order = orderMapper.selectById(command.getOrderId());
        if (order == null || order.getStatus() != 1) {
            throw new BusinessException(40001, "订单状态异常");
        }

        // 2. 调用支付服务
        PaymentRequest request = new PaymentRequest();
        request.setOrderId(order.getOrderId());
        request.setOrderNo(order.getOrderNo());
        request.setAmount(order.getTotalAmount());
        request.setPaymentMethod(command.getPaymentMethod());
        request.setChannel(command.getChannel());

        PaymentResult result = paymentService.pay(request);

        if (!result.isSuccess()) {
            throw new BusinessException(40002, "支付失败：" + result.getMessage());
        }

        // 3. 更新订单状态
        order.setStatus(2); // 已支付
        order.setPaidAmount(order.getTotalAmount());
        order.setPaymentMethod(command.getPaymentMethod());
        order.setPaymentTime(LocalDateTime.now());
        orderMapper.updateById(order);

        // 4. 如果是会员卡订单，激活会员卡
        if (order.getOrderType() == 2) {
            List<OrderDetail> details = orderDetailMapper.selectByOrderId(order.getOrderId());
            for (OrderDetail detail : details) {
                if ("card".equals(detail.getProductType())) {
                    memberCardService.activateCard(
                            detail.getProductId(),
                            order.getMemberId()
                    );
                }
            }
        }

        // 5. 发送支付成功事件
        OrderPaidEvent event = new OrderPaidEvent();
        event.setOrderId(order.getOrderId());
        event.setMemberId(order.getMemberId());
        event.setAmount(order.getTotalAmount());
        messageProducer.send(event);
    }
}
```

```java
// 微信支付实现
@Service("wechatPaymentService")
public class WechatPaymentService implements PaymentService {

    @Override
    public PaymentResult pay(PaymentRequest request) {
        try {
            // 1. 构建微信支付请求
            WxPayUnifiedOrderV3Request wxRequest = new WxPayUnifiedOrderV3Request();
            wxRequest.setOutTradeNo(request.getOrderNo());
            wxRequest.setDescription("教务系统订单");
            wxRequest.setAmount(new WxPayUnifiedOrderV3Request.Amount()
                    .setTotal(request.getAmount().multiply(new BigDecimal("100")).intValue()));
            wxRequest.setPayer(new WxPayUnifiedOrderV3Request.Payer()
                    .setOpenid(getOpenId(request.getMemberId())));

            // 2. 调用微信支付API
            WxPayUnifiedOrderV3Result result = wxPayService.unifiedOrderV3(wxRequest);

            // 3. 保存支付流水
            PaymentFlow flow = new PaymentFlow();
            flow.setPaymentId(idGenerator.nextId());
            flow.setPaymentNo(generatePaymentNo());
            flow.setOrderId(request.getOrderId());
            flow.setMemberId(request.getMemberId());
            flow.setAmount(request.getAmount());
            flow.setPaymentMethod(1); // 微信支付
            flow.setChannel(request.getChannel());
            flow.setTradeNo(result.getPrepayId());
            flow.setStatus(1); // 待支付
            flow.setCreatedAt(LocalDateTime.now());

            paymentFlowMapper.insert(flow);

            // 4. 返回支付参数
            return PaymentResult.builder()
                    .success(true)
                    .paymentNo(flow.getPaymentNo())
                    .qrCode(result.getCodeUrl())
                    .expireTime(calculateExpireTime())
                    .build();

        } catch (Exception e) {
            log.error("微信支付失败", e);
            return PaymentResult.builder()
                    .success(false)
                    .message("支付失败：" + e.getMessage())
                    .build();
        }
    }

    @Override
    @Transactional
    public RefundResult refund(RefundRequest request) {
        try {
            // 1. 获取原支付流水
            PaymentFlow paymentFlow = paymentFlowMapper.selectByOrderId(request.getOrderId());
            if (paymentFlow == null) {
                throw new BusinessException(50001, "原支付记录不存在");
            }

            // 2. 调用微信退款
            WxPayRefundV3Request refundRequest = new WxPayRefundV3Request();
            refundRequest.setOutTradeNo(paymentFlow.getTradeNo());
            refundRequest.setOutRefundNo(generateRefundNo());
            refundRequest.setAmount(new WxPayRefundV3Request.Amount()
                    .setRefund(request.getAmount().multiply(new BigDecimal("100")).intValue())
                    .setTotal(paymentFlow.getAmount().multiply(new BigDecimal("100")).intValue())
                    .setCurrency("CNY"));

            WxPayRefundV3Result refundResult = wxPayService.refundV3(refundRequest);

            // 3. 保存退款流水
            RefundFlow refundFlow = new RefundFlow();
            refundFlow.setRefundId(idGenerator.nextId());
            refundFlow.setRefundNo(generateRefundNo());
            refundFlow.setPaymentId(paymentFlow.getPaymentId());
            refundFlow.setRefundAmount(request.getAmount());
            refundFlow.setStatus(1); // 处理中
            refundFlow.setRefundNo(refundResult.getRefundId());
            refundFlow.setCreatedAt(LocalDateTime.now());

            refundFlowMapper.insert(refundFlow);

            return RefundResult.builder()
                    .success(true)
                    .refundNo(refundFlow.getRefundNo())
                    .status(1)
                    .build();

        } catch (Exception e) {
            log.error("微信退款失败", e);
            throw new BusinessException(50002, "退款失败：" + e.getMessage());
        }
    }
}
```

```java
    @PostMapping("/wechat/notify")
    public String wechatNotify(@RequestBody String body,
                              @RequestHeader("Wechatpay-Signature") String signature,
                              @RequestHeader("Wechatpay-Timestamp") String timestamp,
                              @RequestHeader("Wechatpay-Nonce") String nonce,
                              @RequestHeader("Wechatpay-Serial") String serialNo) {
        try {
            // 1. 验证签名
            if (!verifySignature(body, signature, timestamp, nonce, serialNo)) {
                log.error("微信回调签名验证失败");
                return buildFailResponse();
            }

            // 2. 解析回调数据
            WxPayNotifyV3Result result = wxPayService.parseNotifyV3(body, null);
            WxPayNotifyV3Result.DecryptNotifyResult data = result.getResult();

            // 3. 处理支付结果
            if ("TRANSACTION.SUCCESS".equals(data.getTradeState())) {
                // 更新支付流水状态
                PaymentFlow flow = paymentFlowMapper.selectByTradeNo(data.getOutTradeNo());
                if (flow != null && flow.getStatus() == 1) {
                    flow.setStatus(2); // 支付成功
                    flow.setPaidAt(LocalDateTime.now());
                    flow.setNotifyData(body);
```

### 5.2 API请求封装

```typescript
// src/api/request.ts
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { useAuthStore } from '@/stores/modules/auth'
import router from '@/router'

class Request {
  private instance: AxiosInstance

  constructor() {
    this.instance = axios.create({
      baseURL: import.meta.env.VITE_API_URL,
      timeout: 10000
    })

    this.setupInterceptors()
  }

  private setupInterceptors() {
    // 请求拦截器
    this.instance.interceptors.request.use(
      (config) => {
        const authStore = useAuthStore()
        if (authStore.token) {
          config.headers.Authorization = `Bearer ${authStore.token}`
        }
        return config
      },
      (error) => {
        return Promise.reject(error)
      }
    )

    // 响应拦截器
    this.instance.interceptors.response.use(
      (response: AxiosResponse) => {
        const { code, message } = response.data

        // 请求成功
        if (code === 200) {
          return response.data
        }

        // 业务错误
        ElMessage.error(message || '请求失败')
        return Promise.reject(new Error(message || '请求失败'))
      },
      (error) => {
        const { response } = error

        if (response) {
          switch (response.status) {
            case 401:
              ElMessage.error('登录已过期，请重新登录')
              useAuthStore().logout()
              router.push('/login')
              break
            case 403:
              ElMessage.error('没有权限访问')
              break
            case 404:
              ElMessage.error('请求的资源不存在')
              break
            case 500:
              ElMessage.error('服务器错误')
              break
            default:
              ElMessage.error(response.data?.message || '请求失败')
          }
        } else {
          ElMessage.error('网络错误，请检查网络连接')
        }

        return Promise.reject(error)
      }
    )
  }

  public get<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.get(url, config)
  }

  public post<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.post(url, data, config)
  }

  public put<T = any>(url: string, data?: any, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.put(url, data, config)
  }

  public delete<T = any>(url: string, config?: AxiosRequestConfig): Promise<T> {
    return this.instance.delete(url, config)
  }
}

export default new Request()
```
